SubProceso checkCollision(x, y, map)
    // Define el área del jugador
    Definir width, height Como Entero
    width <- 30
    height <- 50
    
    // Calculamos las esquinas del rectángulo divididas por TILE_SIZE
    Definir left, right, top, bottom Como Entero
    left <- x / TILE_SIZE
    right <- (x + width - 1) / TILE_SIZE
    top <- y / TILE_SIZE
    bottom <- (y + height - 1) / TILE_SIZE
    
    // Verificamos si alguna de las esquinas colisiona con un muro '#'
    Si map[top, left] == '#' O map[top, right] == '#' O map[bottom, left] == '#' O map[bottom, right] == '#' Entonces
        Retornar Verdadero // Hay colisión
    FinSi
    
    Retornar Falso // No hay colisión
FinSubProceso
